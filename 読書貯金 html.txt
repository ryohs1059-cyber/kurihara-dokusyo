    <script>
        // データ保存用のキー（名前を変更しました）
        const STORAGE_KEY = 'kurihara_dokusyo_data';
        
        // データの読み込み
        let books = JSON.parse(localStorage.getItem(STORAGE_KEY)) || [];

        // 初期表示
        updateDisplay();

        // 本を追加する機能
        function addBook() {
            const titleInput = document.getElementById('title-input');
            const pageInput = document.getElementById('page-input');
            const characterImg = document.getElementById('character-img');

            const title = titleInput.value.trim();
            const pages = parseInt(pageInput.value);

            if (!title || isNaN(pages) || pages <= 0) {
                alert('本のタイトルとページ数を正しく入力してね！');
                return;
            }

            // 配列にデータを追加
            const newBook = {
                id: Date.now(),
                title: title,
                pages: pages,
                date: new Date().toLocaleDateString()
            };

            books.unshift(newBook);
            saveData();
            updateDisplay();

            // 入力欄をクリア
            titleInput.value = '';
            pageInput.value = '';
            titleInput.focus();

            // キャラクターのアニメーション発動
            characterImg.classList.remove('bounce');
            void characterImg.offsetWidth; // リフロー
            characterImg.classList.add('bounce');
        }

        // 本を削除する機能
        function deleteBook(id) {
            if(confirm('この記録を消してもいいですか？')) {
                books = books.filter(book => book.id !== id);
                saveData();
                updateDisplay();
            }
        }

        // 画面を更新する機能
        function updateDisplay() {
            const listElement = document.getElementById('history-list');
            const totalElement = document.getElementById('total-count');
            
            listElement.innerHTML = '';
            let totalPages = 0;

            books.forEach(book => {
                totalPages += book.pages;

                const li = document.createElement('li');
                li.className = 'history-item';
                li.innerHTML = `
                    <div>
                        <div class="book-title">${book.title}</div>
                        <div class="book-date">${book.date}</div>
                    </div>
                    <div style="display:flex; align-items:center;">
                        <span class="book-pages">+${book.pages}P</span>
                        <button class="delete-btn" onclick="deleteBook(${book.id})">×</button>
                    </div>
                `;
                listElement.appendChild(li);
            });

            totalElement.textContent = totalPages.toLocaleString();
        }

        // データをブラウザに保存する機能
        function saveData() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(books));
        }
    </script>
